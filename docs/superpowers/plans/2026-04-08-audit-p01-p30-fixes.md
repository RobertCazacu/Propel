```python
def _get_mandatory_missing_for_ai(
    cat_id: str,
    combined_existing: dict[str, str],
    data: "MarketplaceData",
) -> list[str]:
    """Returnează caracteristicile mandatory care lipsesc SAU au valori invalide.

    Diferența față de comportamentul vechi: include și câmpurile cu valori
    prezente dar invalide față de valid_values (restrictive chars).
    """
    mandatory = data.mandatory_chars(cat_id)
    missing = []
    for ch in mandatory:
        val = combined_existing.get(ch)
        if not val:
            missing.append(ch)
            continue
        vs = data.valid_values(cat_id, ch)
        if vs and val not in vs:
            # Valoare prezentă dar invalidă — include pentru re-enrichment
            missing.append(ch)
    return missing
```

```python
def extract_size_from_title(title: str) -> Optional[str]:
    """Extract size from last ' - ' segment of title."""
    if not title:
        return None
    parts = title.rsplit(" - ", 1)
    if len(parts) != 2:
        return None
    return parts[1].strip() or None  # "" → None
```

```python
def complete_structured(self, prompt, schema, system=None):
    import json, re
    system_msg = system or "Returnează DOAR JSON valid, fără text suplimentar."
    raw = self.complete(prompt, max_tokens=500, system=system_msg)
    try:
        text = raw.strip()
        # Încearcă să extragă din ```json ... ``` sau ``` ... ```
        match = re.search(r"```(?:json)?\s*([\s\S]*?)```", text)
        if match:
            text = match.group(1).strip()
        if not text:
            return None
        return json.loads(text)
    except Exception as exc:
        log.debug("complete_structured parse failed: %s | raw: %.200s", exc, raw)
        return None
```
